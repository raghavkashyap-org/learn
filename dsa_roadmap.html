// Power: x^n in O(log n)
long long power(long long x, int n) {
    if (n == 0) return 1;      // base case
    if (n % 2 == 0)               // even power
        return power(x*x, n/2);
    return x * power(x*x, n/2); // odd power
}

// Generate all subsets
void subsets(vector<int>& nums, int i,
             vector<int>& curr,
             vector<vector<int>>& res) {
    if (i == nums.size()) {
        res.push_back(curr); return;
    }
    curr.push_back(nums[i]);   // include
    subsets(nums, i+1, curr, res);
    curr.pop_back();           // exclude (backtrack)
    subsets(nums, i+1, curr, res);
}

// Merge Sort — divide and conquer
void merge(vector<int>& a, int l, int m, int r) {
    vector<int> tmp;
    int i=l, j=m+1;
    while(i<=m && j<=r)
        tmp.push_back(a[i]<=a[j] ? a[i++] : a[j++]);
    while(i<=m) tmp.push_back(a[i++]);
    while(j<=r) tmp.push_back(a[j++]);
    for(int k=l;k<=r;k++) a[k]=tmp[k-l];
}
void mergeSort(vector<int>& a, int l, int r){
    if(l>=r) return;
    int m = (l+r)/2;
    mergeSort(a,l,m); mergeSort(a,m+1,r);
    merge(a,l,m,r);
}

// Classic Binary Search
int binarySearch(vector<int>& arr, int target) {
    int lo = 0, hi = arr.size() - 1;
    while (lo <= hi) {
        int mid = lo + (hi - lo) / 2; // avoid overflow
        if (arr[mid] == target) return mid;
        else if (arr[mid] < target) lo = mid + 1;
        else hi = mid - 1;
    }
    return -1; // not found
}

    int val;
    ListNode* next;
    ListNode(int x) : val(x), next(nullptr) {}
};

// Reverse linked list — iterative
ListNode* reverse(ListNode* head) {
    ListNode* prev = nullptr, *curr = head;
    while (curr) {
        ListNode* nxt = curr->next;
        curr->next = prev;
        prev = curr; curr = nxt;
    }
    return prev;
}

// Height of binary tree
int height(TreeNode* root) {
    if (!root) return 0;
    return 1 + max(height(root->left),
                    height(root->right));
}

// Level order using queue
vector<vector<int>> levelOrder(TreeNode* root) {
    vector<vector<int>> res;
    if(!root) return res;
    queue<TreeNode*> q;
    q.push(root);
    while(!q.empty()) {
        int sz = q.size();
        vector<int> level;
        for(int i=0;i<sz;i++){
            auto node=q.front();q.pop();
            level.push_back(node->val);
            if(node->left) q.push(node->left);
            if(node->right) q.push(node->right);
        }
        res.push_back(level);
    }
    return res;
}

C++
// House Robber — top-down DP
int memo[10001];
int rob(vector<int>& nums, int i) {
    if (i >= nums.size()) return 0;
    if (memo[i] != -1) return memo[i];
    // rob this or skip
    return memo[i] = max(
        nums[i] + rob(nums, i+2),  // rob
        rob(nums, i+1)              // skip
    );
}
